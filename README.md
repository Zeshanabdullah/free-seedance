### Python
This example demonstrates how to use the `requests` library to generate a video using Seedance's free video generator API.

python
import requests

def generate_video(text):
    url = "https://veoaifree.com/seedance-2-0-video-generator-free/"
    payload = {
        "text": text,
        "format": "mp4"
    }
    
    try:
        response = requests.post(url, json=payload)
        response.raise_for_status()  # Raise an error for bad responses
        video_url = response.json().get("video_url")
        print(f"Video generated successfully: {video_url}")
    except requests.exceptions.RequestException as e:
        print(f"An error occurred: {e}")

### PHP
This example shows how to use cURL in PHP to interact with the Seedance API to generate a video.

php
<?php

function generate_video($text) {
    $url = "https://veoaifree.com/seedance-2-0-video-generator-free/";
    $data = json_encode([
        "text" => $text,
        "format" => "mp4"
    ]);

    $ch = curl_init($url);
    curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
    curl_setopt($ch, CURLOPT_POST, true);
    curl_setopt($ch, CURLOPT_HTTPHEADER, [
        'Content-Type: application/json',
        'Content-Length: ' . strlen($data)
    ]);
    curl_setopt($ch, CURLOPT_POSTFIELDS, $data);

    $response = curl_exec($ch);
    if (curl_errno($ch)) {
        echo 'Error: ' . curl_error($ch);
    } else {
        $response_data = json_decode($response, true);
        $video_url = $response_data["video_url"] ?? null;
        echo $video_url ? "Video generated successfully: $video_url" : "Failed to generate video.";
    }

    curl_close($ch);
}

// Example usage
generate_video("Hello, this is a test video generated using Seedance!");


### JavaScript
This example uses the `fetch` API to send a request to the Seedance video generator from a browser or Node.js (ensure node-fetch is installed for Node.js).

javascript
async function generateVideo(text) {
    const url = "https://veoaifree.com/seedance-2-0-video-generator-free/";
    const payload = {
        text: text,
        format: "mp4"
    };

    try {
        const response = await fetch(url, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json'
            },
            body: JSON.stringify(payload)
        });
        
        if (!response.ok) {
            throw new Error('Network response was not ok');
        }

        const data = await response.json();
        const videoUrl = data.video_url;
        console.log(`Video generated successfully: ${videoUrl}`);
    } catch (error) {
        console.error('An error occurred:', error);
    }
}

// Example usage
generateVideo("Hello, this is a test video generated using Seedance!");
